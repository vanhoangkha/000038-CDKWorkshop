1. cdk init --> khởi tạo & download sample app, đồng thời khởi tạo môi trường ảo, khép kín và độc lập với môi trường python bên ngoài	
2. activate.bat --> kích hoạt môi trường ảo hóa 
3. app.py --> Nạp và khởi tạo Instance thuộc lớp CdkWorkshopStack được khai báo trong file cdk_workhop_stack.py
4. cdk_workshop_stack.py --> khởi tạo stack với 2 thực thể con là sqs.Queue và sns.Topic
5. cdk synth --> generate ra một CloudFormation template từ các file cdk_workhop_stack.py
6. cdk ls --> hiển thị danh sách các stack đã có sẵn
7. cdk bootstrap --> Cài đặt resource cần thiết cho cdk toolkit. Bản thân bootstrap cũng là một stack. 
8. cdk deploy cdkworkshop  --> triển khai stack cdkworkshop --> khởi tạo CloudFormation stack 
9. cdk diff --> phát hiện và hiển thị những thay đổi trong file cdk_workhop_stack.py
10. cdk deploy --> áp dụng những thay đổi trên file cdk_workhop_stack.py vào CloudFormation stack 
11. pip install aws-cdk.aws-lambda  --> cài đặt thư viện lambda --> xây dựng ứng dụng với lambda + cdk 
12.  & c:/cdkworkshop/.venv/Scripts/Activate.ps1  --> access vao moi truong virtual environment

#!/usr/bin/env python3 

from aws_cdk import core  --> aws_cdk là bộ thư viện được cài đặt nhờ lệnh "npm -g install aws-cdk". Bộ thư viện sau cài đặt được lưu trong đường dẫn Lib của thư mục chứa application. 
					--> bên trong bộ thư viện aws_cdk chứa rất nhiều thư viện con, trong đó bao gồm cả thư viện Core.
					--> để tránh làm cho ứng dụng trở nên cồng kềnh, ta chỉ thực hiện import một phần của bộ thư viện aws_cdk vào ứng dụng. Cụ thể ở đây, ta chỉ import duy nhất thư viện Core từ bộ thư viện aws_cdk vào ứng dụng

from cdkworkshop.cdkworkshop_stack import CdkWorkshopStack  --> Để import 1 class, thực hiện chỉ rõ đường dẫn tới file chứa class bằng từ khóa "from", và nêu bật class muốn import bằng từ khóa "import". 
												--> Cụ thể, lệnh trên thực hiện import class có tên CdkWorkshopStack nằm trong file cdkworkshop_stack.py có location ở đường dẫn C:\cdkworkshop\cdkworkshop_stack.py


app = core.App()  --> lệnh gọi tới lớp App() nằm trong thư viện Core --> lệnh trả về giá trị của đối tượng app là cấu trúc Construct của App --> gần như khai báo đối tượng dạng Stack
CdkWorkshopStack(app, "cdkworkshop", env={'region': 'us-west-2'})

app.synth()
